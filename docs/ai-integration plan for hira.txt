```sql
-- AI Response Cache for cost optimization
CREATE TABLE IF NOT EXISTS public.ai_response_cache (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  query_hash text UNIQUE NOT NULL,
  response_text text NOT NULL,
  context_type text NOT NULL, -- 'chat', 'meal_parse', 'workout_predict'
  usage_count int DEFAULT 0,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- Index for fast lookups
CREATE INDEX idx_ai_cache_query_hash ON public.ai_response_cache(query_hash);
CREATE INDEX idx_ai_cache_context_type ON public.ai_response_cache(context_type);

-- Auto-update timestamp
CREATE OR REPLACE FUNCTION update_ai_cache_timestamp()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER ai_cache_updated_at
  BEFORE UPDATE ON public.ai_response_cache
  FOR EACH ROW
  EXECUTE FUNCTION update_ai_cache_timestamp();

-- AI Usage Analytics Table
CREATE TABLE IF NOT EXISTS public.ai_usage_analytics (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id uuid REFERENCES auth.users(id) NOT NULL,
  query_type text NOT NULL, -- 'chat', 'meal_parse', 'workout_predict'
  tokens_used int NOT NULL,
  was_cached boolean DEFAULT false,
  latency_ms int,
  cost_usd numeric(10, 6), -- Store actual cost
  created_at timestamptz DEFAULT now()
);

-- Index for analytics queries
CREATE INDEX idx_ai_analytics_user_date ON public.ai_usage_analytics(user_id, created_at);
CREATE INDEX idx_ai_analytics_type ON public.ai_usage_analytics(query_type);

-- RLS for analytics (users can only see their own data)
ALTER TABLE public.ai_usage_analytics ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can view own AI analytics"
  ON public.ai_usage_analytics
  FOR SELECT
  USING (auth.uid() = user_id);

CREATE TABLE IF NOT EXISTS public.ai_usage_limits (
```

```typescript
import Constants from 'expo-constants';

const OPENAI_API_KEY = Constants.expoConfig?.extra?.EXPO_PUBLIC_OPENAI_API_KEY;
const OPENAI_API_URL = 'https://api.openai.com/v1/chat/completions';

interface OpenAIMessage {
  role: 'system' | 'user' | 'assistant';
  content: string;
}

interface OpenAIRequest {
  model: string;
  messages: OpenAIMessage[];
  max_tokens?: number;
  temperature?: number;
  response_format?: { type: 'json_object' };
}

interface OpenAIResponse {
  id: string;
  choices: Array<{
    message: {
      role: string;
      content: string;
    };
    finish_reason: string;
  }>;
  usage: {
    prompt_tokens: number;
    completion_tokens: number;
    total_tokens: number;
  };
}

export class OpenAIService {
  private static calculateCost(promptTokens: number, completionTokens: number): number {
    // GPT-4o Mini pricing: $0.15 per 1M input, $0.60 per 1M output
    const inputCost = (promptTokens / 1_000_000) * 0.15;
    const outputCost = (completionTokens / 1_000_000) * 0.60;
    return inputCost + outputCost;
  }

  static async sendRequest(
    request: OpenAIRequest
  ): Promise<{ response: string; usage: OpenAIResponse['usage']; cost: number }> {
    if (!OPENAI_API_KEY) {
      throw new Error('OpenAI API key not configured');
    }

    const startTime = Date.now();

    try {
      const response = await fetch(OPENAI_API_URL, {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${OPENAI_API_KEY}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify(request),
      });

      if (!response.ok) {
        const error = await response.json();
        throw new Error(`OpenAI API error: ${error.error?.message || 'Unknown error'}`);
      }

      const data: OpenAIResponse = await response.json();
      const latency = Date.now() - startTime;

      const cost = this.calculateCost(
        data.usage.prompt_tokens,
        data.usage.completion_tokens
      );

      console.log(`[OpenAI] Request completed in ${latency}ms, cost: $${cost.toFixed(6)}`);

      return {
        response: data.choices[0].message.content,
        usage: data.usage,
        cost,
      };
    } catch (error) {
      console.error('[OpenAI] Request failed:', error);
      throw error;
    }
  }

  // Helper for chat completions
  static async chat(
    systemPrompt: string,
    userMessage: string,
    options?: {
      maxTokens?: number;
      temperature?: number;
    }
  ) {
    return this.sendRequest({
      model: 'gpt-4o-mini',
      messages: [
        { role: 'system', content: systemPrompt },
        { role: 'user', content: userMessage },
      ],
      max_tokens: options?.maxTokens || 300,
      temperature: options?.temperature || 0.7,
    });
  }

  // Helper for structured JSON output
  static async chatJSON(
    systemPrompt: string,
    userMessage: string,
    options?: {
      maxTokens?: number;
    }
  ) {
    return this.sendRequest({
      model: 'gpt-4o-mini',
      messages: [
        { role: 'system', content: systemPrompt },
        { role: 'user', content: userMessage },
      ],
      max_tokens: options?.maxTokens || 300,
      temperature: 0.3, // Lower temp for structured output
      response_format: { type: 'json_object' },
    });
  }
}
```

```typescript
import { supabase } from '../lib/supabase';
import crypto from 'crypto';

export class AICacheService {
  // Normalize queries to improve cache hit rate
  private static normalizeQuery(query: string): string {
    return query
      .toLowerCase()
      .trim()
      .replace(/\s+/g, ' ') // Normalize whitespace
      .replace(/[^\w\s]/g, ''); // Remove punctuation
  }

  // Generate hash for cache key
  private static generateHash(query: string, contextType: string): string {
    const normalized = this.normalizeQuery(query);
    const key = `${normalized}:${contextType}`;
    return crypto.createHash('md5').update(key).digest('hex');
  }

  // Check if response is cached
  static async getCachedResponse(
    query: string,
    contextType: string
  ): Promise<string | null> {
    try {
      const hash = this.generateHash(query, contextType);

      const { data, error } = await supabase
        .from('ai_response_cache')
        .select('response_text, usage_count')
        .eq('query_hash', hash)
        .single();

      if (error || !data) {
        return null;
      }

      // Increment usage counter
      await supabase
        .from('ai_response_cache')
        .update({ usage_count: data.usage_count + 1 })
        .eq('query_hash', hash);

      console.log(`[AI Cache] HIT for query: "${query}"`);
      return data.response_text;
    } catch (error) {
      console.error('[AI Cache] Error fetching cached response:', error);
      return null;
    }
  }

  // Save response to cache
  static async cacheResponse(
    query: string,
    contextType: string,
    response: string
  ): Promise<void> {
    try {
      const hash = this.generateHash(query, contextType);

      await supabase.from('ai_response_cache').upsert({
        query_hash: hash,
        response_text: response,
        context_type: contextType,
        usage_count: 1,
      });

      console.log(`[AI Cache] SAVED for query: "${query}"`);
    } catch (error) {
      console.error('[AI Cache] Error saving to cache:', error);
    }
  }

  // Clear old cache entries (run periodically)
  static async clearOldEntries(daysOld: number = 30): Promise<void> {
    const cutoffDate = new Date();
    cutoffDate.setDate(cutoffDate.getDate() - daysOld);

    await supabase
      .from('ai_response_cache')
      .delete()
      .lt('created_at', cutoffDate.toISOString());
  }
}
```

```typescript
import { supabase } from '../lib/supabase';

interface AnalyticsEntry {
  userId: string;
  queryType: 'chat' | 'meal_parse' | 'workout_predict';
  tokensUsed: number;
  wasCached: boolean;
  latencyMs: number;
  costUsd: number;
}

export class AIAnalyticsService {
  // Log AI usage
  static async logUsage(entry: AnalyticsEntry): Promise<void> {
    try {
      await supabase.from('ai_usage_analytics').insert({
        user_id: entry.userId,
        query_type: entry.queryType,
        tokens_used: entry.tokensUsed,
        was_cached: entry.wasCached,
        latency_ms: entry.latencyMs,
        cost_usd: entry.costUsd,
      });
    } catch (error) {
      console.error('[AI Analytics] Error logging usage:', error);
    }
  }

  // Get user's today usage
  static async getTodayUsage(userId: string): Promise<number> {
    const today = new Date().toISOString().split('T')[0];

    const { data, error } = await supabase
      .from('ai_usage_analytics')
      .select('id')
      .eq('user_id', userId)
      .gte('created_at', today);

    if (error) {
      console.error('[AI Analytics] Error fetching today usage:', error);
      return 0;
    }

    return data?.length || 0;
  }

  // Get user's cost summary
  static async getCostSummary(userId: string, days: number = 7) {
    const startDate = new Date();
    startDate.setDate(startDate.getDate() - days);

    const { data, error } = await supabase
      .from('ai_usage_analytics')
      .select('cost_usd, was_cached')
      .eq('user_id', userId)
      .gte('created_at', startDate.toISOString());

    if (error || !data) {
      return { totalCost: 0, cachedRequests: 0, totalRequests: 0 };
    }

    const totalCost = data.reduce((sum, entry) => sum + Number(entry.cost_usd), 0);
    const cachedRequests = data.filter(entry => entry.was_cached).length;

    return {
      totalCost,
      cachedRequests,
      totalRequests: data.length,
      cacheHitRate: data.length > 0 ? (cachedRequests / data.length) * 100 : 0,
    };
  }
}
```

```typescript
import { supabase } from '../lib/supabase';
import { OpenAIService } from './openai.service';
import { AICacheService } from './ai-cache.service';
import { AIAnalyticsService } from './ai-analytics.service';
import { AIContextService } from './ai-context.service';

export class AIChatService {
  // Create a new conversation
  static async createConversation(userId: string): Promise<string> {
    const { data, error } = await supabase
      .from('ai_conversations')
      .insert({
        user_id: userId,
        title: 'New conversation',
        context_snapshot: {},
      })
      .select('id')
      .single();

    if (error) throw error;
    return data.id;
  }

  // Build minimal context (not full context)
  private static async buildMinimalContext(
    userId: string,
    userMessage: string
  ): Promise<string> {
    // Determine what context is needed based on the message
    const needsWorkoutContext = /workout|exercise|training|lift/i.test(userMessage);
    const needsNutritionContext = /food|meal|eat|calorie|macro|protein/i.test(userMessage);
    const needsProfileContext = /goal|weight|height|age/i.test(userMessage);

    const contextParts: string[] = [];

    // Only fetch what's needed
    if (needsProfileContext) {
      const profile = await AIContextService.getUserProfile(userId);
      if (profile) {
        contextParts.push(
          `User: ${profile.gender}, goals: ${profile.goals?.join(', ') || 'not set'}`
        );
      }
    }

    if (needsWorkoutContext) {
      const lastWorkout = await AIContextService.getLastWorkout(userId);
      if (lastWorkout) {
        contextParts.push(`Last workout: ${lastWorkout}`);
      }
    }

    if (needsNutritionContext) {
      const todayNutrition = await AIContextService.getTodayNutrition(userId);
      if (todayNutrition) {
        contextParts.push(
          `Today: ${todayNutrition.calories}cal, ${todayNutrition.protein}g protein`
        );
      }
    }

    return contextParts.length > 0 ? contextParts.join('. ') : 'No specific context needed.';
  }

  // Send a chat message
  static async sendChatMessage(
    userId: string,
    conversationId: string,
    userMessage: string
  ): Promise<{ response: string; cost: number; wasCached: boolean }> {
    const startTime = Date.now();

    try {
      // 1. Check cache first
      const cached = await AICacheService.getCachedResponse(userMessage, 'chat');
      if (cached) {
        // Log cached usage
        await AIAnalyticsService.logUsage({
          userId,
          queryType: 'chat',
          tokensUsed: 0,
          wasCached: true,
          latencyMs: Date.now() - startTime,
          costUsd: 0,
        });

        // Save message to conversation
        await this.saveMessage(conversationId, 'user', userMessage);
        await this.saveMessage(conversationId, 'assistant', cached);

        return { response: cached, cost: 0, wasCached: true };
      }

      // 2. Build minimal context
      const context = await this.buildMinimalContext(userId, userMessage);

      // 3. Call OpenAI
      const systemPrompt = `You are Hira, a friendly and knowledgeable fitness and wellness coach. 
You help users with workouts, nutrition, and healthy habits. Keep responses concise and actionable.

Context: ${context}`;

      const { response, usage, cost } = await OpenAIService.chat(
        systemPrompt,
        userMessage,
        { maxTokens: 300 }
      );

      // 4. Save to cache
      await AICacheService.cacheResponse(userMessage, 'chat', response);

      // 5. Log analytics
      await AIAnalyticsService.logUsage({
        userId,
        queryType: 'chat',
        tokensUsed: usage.total_tokens,
        wasCached: false,
        latencyMs: Date.now() - startTime,
        costUsd: cost,
      });

      // 6. Save messages to DB
      await this.saveMessage(conversationId, 'user', userMessage);
      await this.saveMessage(conversationId, 'assistant', response);

      return { response, cost, wasCached: false };
    } catch (error) {
      console.error('[AI Chat] Error:', error);
      throw error;
    }
  }

  // Save a message to the conversation
  private static async saveMessage(
    conversationId: string,
    role: 'user' | 'assistant',
    content: string
  ): Promise<void> {
    await supabase.from('ai_messages').insert({
      conversation_id: conversationId,
      role,
      content,
    });
  }

  // Get conversation messages
  static async getMessages(conversationId: string) {
    const { data, error } = await supabase
      .from('ai_messages')
      .select('*')
      .eq('conversation_id', conversationId)
      .order('created_at', { ascending: true });

    if (error) throw error;
    return data;
  }
}
```

```typescript
import Constants from 'expo-constants';
import { supabase } from '../lib/supabase';
import crypto from 'crypto';

const OPENAI_API_KEY = Constants.expoConfig?.extra?.EXPO_PUBLIC_OPENAI_API_KEY;

interface WorkoutSuggestion {
  muscleGroup: string;
  workoutType: string;
  reasoning: string;
  estimatedDuration: number;
}

interface ExerciseTip {
  exercise: string;
  keyPoints: string[];
  commonMistakes: string[];
  proTip: string;
}

export class AIWorkoutService {
  // === COST CALCULATION ===
  private static calculateCost(inputTokens: number, outputTokens: number): number {
    const inputCost = (inputTokens / 1_000_000) * 0.15;
    const outputCost = (outputTokens / 1_000_000) * 0.60;
    return inputCost + outputCost;
  }

  // === CACHE HELPERS ===
  private static generateCacheKey(type: string, data: any): string {
    const str = JSON.stringify({ type, data });
    return crypto.createHash('md5').update(str).digest('hex');
  }

  private static async checkCache(cacheKey: string): Promise<any | null> {
    const { data } = await supabase
      .from('ai_workout_cache')
      .select('response_data, hit_count')
      .eq('cache_key', cacheKey)
      .single();

    if (data) {
      // Increment hit counter
      await supabase
        .from('ai_workout_cache')
        .update({ hit_count: data.hit_count + 1 })
        .eq('cache_key', cacheKey);

      console.log(`[AI Workout] Cache HIT: ${cacheKey.substring(0, 8)}`);
      return data.response_data;
    }

    return null;
  }

  private static async saveToCache(
    cacheKey: string,
    cacheType: string,
    responseData: any
  ): Promise<void> {
    await supabase.from('ai_workout_cache').upsert({
      cache_key: cacheKey,
      cache_type: cacheType,
      response_data: responseData,
      hit_count: 1,
    });
  }

  private static async logUsage(
    userId: string,
    featureType: string,
    tokensUsed: number,
    wasCached: boolean,
    costUsd: number
  ): Promise<void> {
    await supabase.from('ai_workout_usage').insert({
      user_id: userId,
      feature_type: featureType,
      tokens_used: tokensUsed,
      was_cached: wasCached,
      cost_usd: costUsd,
    });
  }

  // === OPENAI CALL ===
  private static async callOpenAI(
    systemPrompt: string,
    userPrompt: string,
    maxTokens: number = 200
  ): Promise<{ response: string; inputTokens: number; outputTokens: number }> {
    const response = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: {
        Authorization: `Bearer ${OPENAI_API_KEY}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: 'gpt-4o-mini',
        messages: [
          { role: 'system', content: systemPrompt },
          { role: 'user', content: userPrompt },
        ],
        max_tokens: maxTokens,
        temperature: 0.7,
        response_format: { type: 'json_object' },
      }),
    });

    if (!response.ok) {
      throw new Error(`OpenAI API error: ${response.statusText}`);
    }

    const data = await response.json();
    return {
      response: data.choices[0].message.content,
      inputTokens: data.usage.prompt_tokens,
      outputTokens: data.usage.completion_tokens,
    };
  }

  // === FEATURE 1: SMART WORKOUT SUGGESTION ===
  static async suggestWorkout(
    userId: string,
    recentWorkouts: Array<{ muscleGroup: string; date: string }>
  ): Promise<WorkoutSuggestion> {
    // Build cache key
    const cacheKey = this.generateCacheKey('suggestion', {
      recent: recentWorkouts.map(w => w.muscleGroup),
    });

    // Check cache first
    const cached = await this.checkCache(cacheKey);
    if (cached) {
      await this.logUsage(userId, 'suggestion', 0, true, 0);
      return cached;
    }

    // Build context
    const recentList = recentWorkouts
      .map(w => `${w.muscleGroup} (${w.date})`)
      .join(', ');

    const systemPrompt = `You are a smart workout coach. Suggest the next best workout based on muscle group rotation and recovery.
Return ONLY valid JSON with this exact structure:
{
  "muscleGroup": "chest|back|legs|shoulders|arms|fullbody",
  "workoutType": "strength|hypertrophy|endurance",
  "reasoning": "brief explanation",
  "estimatedDuration": 45
}`;

    const userPrompt = `Recent workouts: ${recentList || 'none'}. What should I train next?`;

    // Call OpenAI
    const { response, inputTokens, outputTokens } = await this.callOpenAI(
      systemPrompt,
      userPrompt,
      150
    );

    const suggestion: WorkoutSuggestion = JSON.parse(response);
    const cost = this.calculateCost(inputTokens, outputTokens);

    // Save to cache and log
    await this.saveToCache(cacheKey, 'workout_suggestion', suggestion);
    await this.logUsage(userId, 'suggestion', inputTokens + outputTokens, false, cost);

    console.log(`[AI Workout] Suggestion generated - Cost: $${cost.toFixed(6)}`);
    return suggestion;
  }

  // === FEATURE 2: EXERCISE FORM TIPS ===
  static async getExerciseTip(userId: string, exerciseName: string): Promise<ExerciseTip> {
    // Normalize exercise name for caching
    const normalized = exerciseName.toLowerCase().trim();
    const cacheKey = this.generateCacheKey('exercise_tip', { exercise: normalized });

    // Check cache
    const cached = await this.checkCache(cacheKey);
    if (cached) {
      await this.logUsage(userId, 'exercise_tip', 0, true, 0);
      return cached;
    }

    const systemPrompt = `You are a strength coach. Provide concise form tips for exercises.
Return ONLY valid JSON with this exact structure:
{
  "exercise": "exercise name",
  "keyPoints": ["point 1", "point 2", "point 3"],
  "commonMistakes": ["mistake 1", "mistake 2"],
  "proTip": "one advanced tip"
}`;

    const userPrompt = `Give me form tips for: ${exerciseName}`;

    const { response, inputTokens, outputTokens } = await this.callOpenAI(
      systemPrompt,
      userPrompt,
      200
    );

    const tip: ExerciseTip = JSON.parse(response);
    const cost = this.calculateCost(inputTokens, outputTokens);

    await this.saveToCache(cacheKey, 'exercise_tip', tip);
    await this.logUsage(userId, 'exercise_tip', inputTokens + outputTokens, false, cost);

    console.log(`[AI Workout] Exercise tip generated - Cost: $${cost.toFixed(6)}`);
    return tip;
  }

  // === FEATURE 3: PROGRAM GENERATION HELPER ===
  static async generateProgramWeek(
    userId: string,
    goal: string,
    daysPerWeek: number,
    fitnessLevel: string
  ): Promise<Array<{ day: number; focus: string; exercises: string[] }>> {
    const cacheKey = this.generateCacheKey('program_week', {
      goal,
      days: daysPerWeek,
      level: fitnessLevel,
    });

    const cached = await this.checkCache(cacheKey);
    if (cached) {
      await this.logUsage(userId, 'program_gen', 0, true, 0);
      return cached;
    }

    const systemPrompt = `You are a program designer. Create a weekly workout split.
Return ONLY valid JSON array with this structure:
[
  {
    "day": 1,
    "focus": "chest & triceps",
    "exercises": ["bench press", "incline dumbbell press", "tricep dips"]
  }
]`;

    const userPrompt = `Create a ${daysPerWeek}-day split for ${goal} at ${fitnessLevel} level.`;

    const { response, inputTokens, outputTokens } = await this.callOpenAI(
      systemPrompt,
      userPrompt,
      300
    );

    const program = JSON.parse(response);
    const cost = this.calculateCost(inputTokens, outputTokens);

    await this.saveToCache(cacheKey, 'program_help', program);
    await this.logUsage(userId, 'program_gen', inputTokens + outputTokens, false, cost);

    console.log(`[AI Workout] Program generated - Cost: $${cost.toFixed(6)}`);
    return program;
  }
}
```

```typescript
  // Becomes: recent: ['chest', 'back']

// generateCacheKey() does this:
private static generateCacheKey(type: string, data: any): string {
  const str = JSON.stringify({ type, data });
  // str = '{"type":"suggestion","data":{"recent":["chest","back"]}}'
  
  return crypto.createHash('md5').update(str).digest('hex');
  // Returns: "a3f8e92b1c4d..." (32 character hash)

// Step 3.2: Check if this exact question was asked before

// checkCache() does this:
private static async checkCache(cacheKey: string): Promise<any | null> {
  // Query Supabase
  const { data } = await supabase
    .from('ai_workout_cache')
    .select('response_data, hit_count')
    .eq('cache_key', cacheKey)
    .single();
  
  if (data) {
    // Found it! This question was answered before
    
    // Increment hit counter (for analytics)
    await supabase
      .from('ai_workout_cache')
      .update({ hit_count: data.hit_count + 1 })
      .eq('cache_key', cacheKey);
    
    console.log('✅ Cache HIT - returning instant response');
    
    // Return the cached answer (no API call needed!)
    return data.response_data;
    // Returns: {
    //   muscleGroup: 'legs',
    //   reasoning: 'You did upper body last 2 sessions',

    // }

// If cached response exists:

  await this.logUsage(userId, 'suggestion', 0, true, 0);

  return cached;
}
```

**Why separate system and user prompts?**

- **System prompt**: Sets AI's personality and output format (AI reads this first)
- **User prompt**: The actual question with context (AI answers this)

This separation gives better results than combining them.

```typescript
  userPrompt: string,
  maxTokens: number = 200
) {
  // Build the HTTP request
  const response = await fetch('https://api.openai.com/v1/chat/completions', {
    method: 'POST',
    headers: {
      // Your API key for authentication
      'Authorization': `Bearer ${OPENAI_API_KEY}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      // Which model to use
      model: 'gpt-4o-mini',
      
      // The conversation messages
      messages: [
        { 
          role: 'system', 
          content: systemPrompt 
        },
        { 
          role: 'user', 
          content: userPrompt 
        }
      ],
      
      // Limit response length (controls cost)
      max_tokens: maxTokens,
      
      // Creativity level (0-2, lower = more consistent)
      temperature: 0.7,
      
      // Force JSON output (no extra text)
      response_format: { type: 'json_object' }
    })
  });
  
  // [REQUEST SENT - Now in STAGE 6]
}
```

### **Step 2: Create Context Builder**

```typescript
import { supabase } from '../../lib/supabase';

interface UserContext {
  age?: number;
  gender?: string;
  fitnessLevel?: string;
  goals?: string[];
  recentWorkouts: Array<{
    muscleGroup: string;
    date: string;
    type: string;
  }>;
  lastWeekVolume?: string;
}

export class ContextBuilder {
  // Build a concise "user memo" instead of raw data dump
  static buildUserMemo(context: UserContext): string {
    const parts: string[] = [];

    // Basic profile
    if (context.age && context.gender) {
      parts.push(`USER: ${context.age}yo ${context.gender}`);
    }

    // Fitness level
    if (context.fitnessLevel) {
      parts.push(`LEVEL: ${context.fitnessLevel}`);
    }

    // Goals
    if (context.goals && context.goals.length > 0) {
      parts.push(`GOALS: ${context.goals.join(', ')}`);
    }

    // Recent training
    if (context.recentWorkouts.length > 0) {
      const recent = context.recentWorkouts
        .slice(0, 3)
        .map(w => `${w.muscleGroup} (${w.date})`)
        .join(', ');
      parts.push(`RECENT TRAINING: ${recent}`);
    } else {
      parts.push(`RECENT TRAINING: None logged`);
    }

    // Training volume assessment
    if (context.lastWeekVolume) {
      parts.push(`VOLUME: ${context.lastWeekVolume}`);
    }

    return parts.join('\n');
  }

  // Fetch and build full context from database
  static async buildFullContext(userId: string): Promise<UserContext> {
    // Get profile
    const { data: profile } = await supabase
      .from('profiles')
      .select('date_of_birth, gender')
      .eq('id', userId)
      .single();

    // Get health profile
    const { data: health } = await supabase
      .from('user_health_profile')
      .select('goals, activity_level')
      .eq('user_id', userId)
      .single();

    // Get recent workouts (last 7 days)
    const sevenDaysAgo = new Date();
    sevenDaysAgo.setDate(sevenDaysAgo.getDate() - 7);

    const { data: workouts } = await supabase
      .from('workout_sessions')
      .select('session_type, performed_at, duration_minutes')
      .eq('user_id', userId)
      .gte('performed_at', sevenDaysAgo.toISOString())
      .order('performed_at', { ascending: false })
      .limit(5);

    // Calculate age
    let age;
    if (profile?.date_of_birth) {
      const birthDate = new Date(profile.date_of_birth);
      const today = new Date();
      age = today.getFullYear() - birthDate.getFullYear();
    }

    // Map activity level to fitness level
    const fitnessLevelMap: Record<string, string> = {
      sedentary: 'beginner',
      light: 'beginner',
      moderate: 'intermediate',
      active: 'intermediate',
      athlete: 'advanced',
    };

    const fitnessLevel = health?.activity_level
      ? fitnessLevelMap[health.activity_level]
      : 'intermediate';

    // Assess training volume
    let lastWeekVolume = 'low';
    if (workouts && workouts.length >= 5) {
      lastWeekVolume = 'high';
    } else if (workouts && workouts.length >= 3) {
      lastWeekVolume = 'moderate';
    }

    return {
      age,
      gender: profile?.gender,
      fitnessLevel,
      goals: health?.goals || [],
      recentWorkouts:
        workouts?.map(w => ({
          muscleGroup: w.session_type || 'general',
          date: new Date(w.performed_at).toLocaleDateString(),
          type: 'strength',
        })) || [],
      lastWeekVolume,
    };
  }

  // Build minimal context (for cheaper requests)
  static async buildMinimalContext(userId: string): Promise<string> {
    const { data: workouts } = await supabase
      .from('workout_sessions')
      .select('session_type, performed_at')
      .eq('user_id', userId)
      .order('performed_at', { ascending: false })
      .limit(3);

    if (!workouts || workouts.length === 0) {
      return 'USER: New user, no workout history';
    }

    const recent = workouts
      .map(w => `${w.session_type || 'general'} (${new Date(w.performed_at).toLocaleDateString()})`)
      .join(', ');

    return `RECENT TRAINING: ${recent}`;
  }
}
```

```typescript
    // Check cache
    const cached = await this.checkCache(cacheKey);
    if (cached) {
      await this.logUsage(userId, 'suggestion', 0, true, 0);
      return cached;
    }

    // Build the full system prompt
    const systemPrompt = `${HIRA_MASTER_SYSTEM_PROMPT}

${WORKOUT_SUGGESTION_FOCUS}`;

    // Build user prompt with context
    const userPrompt = `${userMemo}

USER QUESTION: What should I train today?

ADDITIONAL CONTEXT:
- If I've trained the same muscle group in the last 48 hours, suggest something else
- Consider my training volume this week
- Suggest exercises appropriate for my fitness level`;

    // Call OpenAI
    const { response, inputTokens, outputTokens } = await this.callOpenAI(
      systemPrompt,
      userPrompt,
      300 // Increased token limit for richer responses
    );

    const aiResponse: HiraAIResponse = JSON.parse(response);
    const cost = this.calculateCost(inputTokens, outputTokens);

    // Save and log
    await this.saveToCache(cacheKey, 'workout_suggestion_v2', aiResponse);
    await this.logUsage(userId, 'suggestion', inputTokens + outputTokens, false, cost);

    console.log(`[Hira AI] Workout suggested - Cost: $${cost.toFixed(6)}`);
    return aiResponse;
  }

  // === UPGRADED EXERCISE TIP ===
  static async getExerciseTip(userId: string, exerciseName: string): Promise<HiraAIResponse> {
    const normalized = exerciseName.toLowerCase().trim();
    const cacheKey = this.generateCacheKey('form_tip_v2', { exercise: normalized });

    const cached = await this.checkCache(cacheKey);
    if (cached) {
      await this.logUsage(userId, 'exercise_tip', 0, true, 0);
      return cached;
    }

    // Get minimal context (fitness level only)
    const context = await ContextBuilder.buildMinimalContext(userId);

    const systemPrompt = `${HIRA_MASTER_SYSTEM_PROMPT}

${FORM_TIP_FOCUS}`;

    const userPrompt = `${context}

USER QUESTION: Give me form tips for ${exerciseName}

Focus on the most critical cues for proper execution.`;

    const { response, inputTokens, outputTokens } = await this.callOpenAI(
      systemPrompt,
      userPrompt,
      250
    );

    const aiResponse: HiraAIResponse = JSON.parse(response);
    const cost = this.calculateCost(inputTokens, outputTokens);

    await this.saveToCache(cacheKey, 'form_tip_v2', aiResponse);
    await this.logUsage(userId, 'exercise_tip', inputTokens + outputTokens, false, cost);

    return aiResponse;
  }

  // === NEW: GENERAL CHAT (for Hira Chat screen) ===
  static async chatWithHira(
    userId: string,
    userMessage: string,
    conversationHistory?: Array<{ role: string; content: string }>
  ): Promise<HiraAIResponse> {
    // Check if this is a common question (cache it)
    const normalizedMessage = userMessage.toLowerCase().trim();
    const cacheKey = this.generateCacheKey('chat_v2', { msg: normalizedMessage });

    const cached = await this.checkCache(cacheKey);
    if (cached) {
      await this.logUsage(userId, 'chat', 0, true, 0);
      return cached;
    }

    // Build context
    const context = await ContextBuilder.buildMinimalContext(userId);

    const systemPrompt = HIRA_MASTER_SYSTEM_PROMPT;

    // Include conversation history if available
    const messages: Array<{ role: string; content: string }> = [
      { role: 'system', content: systemPrompt },
    ];

    // Add history (last 3 messages only to save tokens)
    if (conversationHistory) {
      messages.push(
        ...conversationHistory.slice(-3).map(msg => ({
          role: msg.role,
          content: msg.content,
        }))
      );
    }

    // Add current message with context
    messages.push({
      role: 'user',
      content: `${context}

USER QUESTION: ${userMessage}`,
    });

    // Call OpenAI with conversation
    const openAIResponse = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: {
        Authorization: `Bearer ${process.env.OPENAI_API_KEY}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: 'gpt-4o-mini',
        messages,
        max_tokens: 400,
        temperature: 0.7,
        response_format: { type: 'json_object' },
      }),
    });

    const data = await openAIResponse.json();
    const aiResponse: HiraAIResponse = JSON.parse(data.choices[0].message.content);
    const cost = this.calculateCost(data.usage.prompt_tokens, data.usage.completion_tokens);

    // Only cache if it's a general question (not user-specific conversation)
    if (!conversationHistory || conversationHistory.length === 0) {
      await this.saveToCache(cacheKey, 'chat_v2', aiResponse);
    }

    await this.logUsage(userId, 'chat', data.usage.total_tokens, false, cost);

    return aiResponse;
  }
}
```
